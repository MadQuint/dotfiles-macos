# direnv Configuration

# Create a .envrc file in any directory to auto-load environment variables
# Example .envrc:
#   export NODE_ENV=development
#   export DATABASE_URL=postgres://localhost/mydb

# Allow the .envrc file with: direnv allow .

# Common patterns:
# 1. Project-specific env vars
# 2. Auto-activate virtual environments
# 3. Load secrets from 1Password/Keychain
# 4. Set PATH for project binaries

# To create .envrc in a project:
#   echo 'export PROJECT=myproject' > .envrc
#   direnv allow
